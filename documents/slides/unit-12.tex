\begin{frame}
\textbf{magnet}: \uncover<2->{something that can attract (or repel) metal or another magnet}

\pause

\pause

\begin{itemize}[<+->]
    \item has two poles: north and south
    \item if divided, each part has a north and south pole
    \item exerts magnetic force
    \item opposite poles attract, like poles repel
\end{itemize}
