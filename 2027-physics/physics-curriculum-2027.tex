\textbf{Mathematical Formulas}

% Students must memorize the four core kinematic equations for constant acceleration and identify which variables are present and missing in each:



\begin{itemize}[itemsep=0pt,topsep=-3pt]
    \item \textbf{Know} the variables used to represent physical quantities, including knowing that $x$ stands position, $\Delta x$ stands for displacement, $t$ stands for time, $v$ stands for velocity.
    \item \textbf{Know} that the subscripts $i$ and $f$ are used to denote initial and final variables, respectively (e.g.,\\$v_i$ = initial velocity, $x_f$ = final position, etc.).
    \item \textbf{Know} that the base SI unit of position is the meter (m), the base SI unit of time is the second (s), and the base SI unit of velocity is the meter per second (m/s).
    \item \textbf{Know} that average speed is given by distance divided by time.
    \item \textbf{Know} that average velocity is given by $\displaystyle \bar{v} = \frac{\Delta x}{\Delta t}$.
    \item \textbf{Know} that slope is rise over run.
\end{itemize}

\section{Acceleration}

\textbf{Vocabulary}

\begin{itemize}[itemsep=0pt,topsep=-3pt]
    \item Define \textbf{\gls{acceleration}} as \glsdesc{acceleration}; or as \glsdesc{acceleration2}.
    \item Define \textbf{average acceleration} as \glsdesc{average acceleration}.
    \item Define \textbf{acceleration vs. time graph} as \glsdesc{acceleration vs. time graph}.
\end{itemize}

\medskip

\textbf{Core Principles}

% Students must articulate the following physical rules governing one-dimensional motion:

\begin{itemize}[itemsep=0pt,topsep=-3pt]
    \item \textbf{Recall} that $x$ stands for position, $v$ stands for velocity, and $t$ stands for time.
    \item \textbf{Recall} that the position axis conventionally determines the sign (direction) of velocity and acceleration vectors.
    \item \textbf{Know} that an object is accelerating if its velocity is changing.
    \item \textbf{Recognize} that this unit (and course) deals only with constant (uniform) acceleration.
    \item \textbf{Know} that acceleration is a vector, so that the sign ($+$ or $-$) of acceleration indicates the direction.
    \item \textbf{Understand} that an object is speeding up if its velocity and acceleration vectors point in the same direction (both positive or both negative).
    \item \textbf{Understand} that an object is slowing down if its velocity and acceleration vectors point in opposite directions (one is positive, the other is negative).
    \item \textbf{Understand} that constant velocity implies that acceleration is exactly zero.
    \item \textbf{Understand} that an object can have a velocity of zero but still be accelerating (e.g., at the peak of a thrown ball's trajectory).
    \item \textbf{Understand} that, if an object is accelerating, the velocity vs.\,time graph has a slope that is constant.
    \item \textbf{Know} that the slope of a velocity vs.\,time graph is acceleration.
    \item \textbf{Understand} that, if an object is accelerating, the position vs.\,time graph has a slope that is changing.
    \item \textbf{Know} that the area bounded by a line in a velocity vs.\,time graph is displacement.
    \item \textbf{Know} that the area bounded by a line in an acceleration vs.\,time graph is change in velocity.
\end{itemize}

\medskip

\textbf{Learning Goals}

\begin{enumerate}[itemsep=0pt,topsep=-3pt]
    \item \textbf{Justify} whether or not an object is accelerating given a data table of velocity and time.
    \item \textbf{Justify} whether or not an object is accelerating given a data table of position and time.
    \item \textbf{Draw} a motion map for an object undergoing positive or negative acceleration.
    \item \textbf{Identify} whether or not an object is accelerating from a velocity vs.\,time graph.
    \item \textbf{Identify} the direction of acceleration from a velocity vs.\,time graph.
    \item \textbf{Identify} the sign (positive or negative) of the slope of a point in a position vs.\,time graph, and associate it with the direction of the object's velocity.
    % \item \textbf{Identify} whether an object is speeding up or slowing down from a positive vs.\,time graph.
    \item \textbf{Estimate} changes (increasing or decreasing) in slope across time in a position vs.\,time, to \textbf{determine} (a) whether an object is speeding up or slowing down and (b) the sign of acceleration.
    \item \textbf{Pair} position vs.\,time and velocity vs.\,time graphs of a common object undergoing acceleration.
    \item \textbf{Calculate} change in velocity given a particle's initial and final velocities.
    \item \textbf{Calculate} average acceleration given a particle's initial and final velocities and the time of travel.
    \item \textbf{Calculate} average acceleration by finding the slope in a velocity vs.\,time graph.

\bigskip

\textbf{Mathematical Formulas}

\begin{itemize}[itemsep=0pt,topsep=-3pt]
    \item \textbf{Know} variables used to represent physical quantities, including: $a$ = acceleration; $\Delta v$ = change in velocity; $\Delta t$ = change in time.
    \item \textbf{Know} that the base SI unit of acceleration is the meter per second per second (m/s/s), or the meter per second squared ($\mathrm{m/s^2}$).
    \item \textbf{Know} that average acceleration is given by $\displaystyle a = \frac{\Delta v}{\Delta t}$.
    \item \textbf{Know} that the area of a triangle is given by $\frac{1}{2}\times \text{base} \times \text{height}$.
\end{itemize}

\textbf{Mathematical Formulas}

\begin{itemize}[topsep=-3pt,itemsep=0pt]
    \item \textbf{Know} velocity as a function of time: $\displaystyle v_f = v_i + at$
    \item \textbf{Know} position as a function of time: $\displaystyle \Delta x = v_i t + \frac{1}{2}at^2$
    \item \textbf{Know} velocity as a function of position: $\displaystyle v_f^2 = v_i^2 + 2a\Delta x$
    \item \textbf{Know} average velocity to find displacement: $\displaystyle \Delta x = \frac{1}{2}(v_i + v_f)t$
\end{itemize}
