\subsection*{Fundamental Principles} %...THE CONCEPTUAL RULES

\begin{itemize}[itemsep=0pt,topsep=-3pt]
    \item \textbf{Know} that a vector is graphically represented as an arrow.
    \item \textbf{Know} that a vector is positive if it is directed to the right, and negative if it is directed to the left.
    \item \textbf{Know} that position, velocity, and displacement are vectors.
    \item \textbf{Know} that distance, speed, and time are scalars.
    \item \textbf{Know} that positive direction for position or velocity is to the right on a horizontal position axis, and negative direction is to the left.
    \begin{center}
        \begin{tikzpicture}[y=1cm,x=1cm]
            \foreach \x in {-5,-4,...,5} {

            \draw[->,ultra thick] (-1.5,0.4) -- (1.5,0.4) node[above,pos=0.5] {positive direction};
            \draw[->,ultra thick] (1.5,1.2) -- (-1.5,1.2) node[above,pos=0.5] {negative direction};
            % \draw[thick,<-] (0,0.15) -- ++(0,0.5);
        \end{tikzpicture}
    \end{center}

    \item \textbf{Identify} the origin in a position reference frame.
    \item \textbf{Know} that an object's position vector starts at the origin (at zero) and ends at the object's location on the position axis.
    \item \textbf{Understand} that an object is in motion if its position is changing.

    \begin{center}
        \begin{tikzpicture}[y=1cm,x=1cm]
            \foreach \x in {-5,-4,...,5} {
                \draw[ultra thin] (\x,-0.09) -- (\x,0.09) node[below=0.2] {$\x$};

            \draw[thick,->] (2.8,0.8) -- ++(0.6,0) node[above,pos=0.5] {$\vec{v}$};
        \end{tikzpicture}
    \end{center}
    \item \textbf{Understand} that constant velocity implies that displacement is constant across equal time intervals.

    \begin{figure}[H]
    \centering
    \begin{tikzpicture}[y=3.8mm,x=4mm]

        }
    \end{tikzpicture}

    \label{C0fgZ}
    \end{figure}

    \item \textbf{Understand} that a vector's direction is indicated by a positive ($+$) or negative ($-$) sign\\(e.g., $\vec{v} = \SI{-3}{m/s}$,  $\vec{x} = \SI{8}{m}$).
    
    \item \textbf{Understand} that an object is moving away from the origin if its position and velocity vectors point in the same directions (both positive or both negative).

    \begin{center}
        \begin{tikzpicture}[y=1cm,x=1cm]
            \foreach \x in {-5,-4,...,5} {
                \draw[ultra thin] (\x,-0.09) -- (\x,0.09) node[below=0.2] {$\x$};
            }
            \draw[->] (-5,0) -- (5,0) node[below=0.6,pos=0.5] {Position (m)};
            \draw[thick,->] (0,0.3) -- ++(-1.9,0) node[above,pos=0.5] {$\vec{x}$};
            \node[above] at (-2,0) {\Strichmaxerl[2]};
            \draw[very thick,->] (-2,0.7) -- ++(-1,0) node[above,pos=0.5] {$\vec{v}$};
        \end{tikzpicture}
    \end{center}
    \item \textbf{Understand} that an object is moving towards the origin if its position and velocity vectors point in opposite directions (one is positive, the other is negative).
    \item \textbf{Distinguish} between distance (a scalar) and displacement (a vector).
    \item \textbf{Understand} that an object can have a position of zero but still be moving (e.g., at the origin of the reference frame).
    \item \textbf{Distinguish} between speed (a scalar) and velocity (a vector).
\end{itemize}

\subsection*{Mathematical Formulas} %...THE MATHEMATICAL RULES

\begin{itemize}[itemsep=0pt,topsep=-3pt]
    \item \textbf{Know} the variables used to represent physical quantities, including knowing that $x$ stands position, $\Delta x$ stands for displacement, $t$ stands for time, $v$ stands for velocity.
    \item \textbf{Know} that the subscripts $i$ and $f$ are used to denote initial and final variables, respectively (e.g.,\\$v_i$ = initial velocity, $x_f$ = final position, etc.).
    \item \textbf{Know} that the base SI unit of position is the meter (m), the base SI unit of time is the second (s), and the base SI unit of velocity is the meter per second (m/s).
    \item \textbf{Know} that average speed is given by distance divided by time.
    \item \textbf{Know} that average velocity is given by $\displaystyle \bar{v} = \frac{\Delta x}{\Delta t}$.
    \item \textbf{Know} that slope is rise over run.
    \item \textbf{Know} that the area of a rectangle is base $\times$ height.
\end{itemize}

\subsection*{Fundamental Principles}


\begin{itemize}[itemsep=0pt,topsep=-3pt]
    \item \textbf{Recall} that $x$ stands for position, $v$ stands for velocity, and $t$ stands for time.
    \item \textbf{Recall} that the position axis conventionally determines the sign (direction) of velocity and acceleration vectors.
    \item \textbf{Know} that an object is accelerating if its velocity is changing.
    \item \textbf{Recognize} that this unit (and course) deals only with constant (uniform) acceleration.
    \item \textbf{Know} that acceleration is a vector, so that the sign ($+$ or $-$) of acceleration indicates the direction.
    \item \textbf{Understand} that an object is speeding up if its velocity and acceleration vectors point in the same direction (both positive or both negative).
    \item \textbf{Understand} that an object is slowing down if its velocity and acceleration vectors point in opposite directions (one is positive, the other is negative).
    \item \textbf{Understand} that constant velocity implies that acceleration is exactly zero.
    \item \textbf{Understand} that an object can have a velocity of zero but still be accelerating (e.g., at the peak of a thrown ball's trajectory).
    \item \textbf{Understand} that, if an object is accelerating, the velocity vs.\,time graph has a slope that is constant.
    \item \textbf{Know} that the slope of a velocity vs.\,time graph is acceleration.
    \item \textbf{Understand} that, if an object is accelerating, the position vs.\,time graph has a slope that is changing.
    \item \textbf{Know} that the area bounded by a line in a velocity vs.\,time graph is displacement.
    \item \textbf{Know} that the area bounded by a line in an acceleration vs.\,time graph is change in velocity.
\end{itemize}

\subsection*{Mathematical Formulas}

\begin{itemize}[itemsep=0pt,topsep=-3pt]
    \item \textbf{Know} variables used to represent physical quantities, including: $a$ = acceleration; $\Delta v$ = change in velocity; $\Delta t$ = change in time.
    \item \textbf{Know} that the base SI unit of acceleration is the meter per second per second (m/s/s), or the meter per second squared ($\mathrm{m/s^2}$).
    \item \textbf{Know} that average acceleration is given by $\displaystyle a = \frac{\Delta v}{\Delta t}$.
    \item \textbf{Know} that the area of a triangle is given by $\frac{1}{2}\times \text{base} \times \text{height}$.
\end{itemize}

\subsection*{Fundamental Principles}

\begin{itemize}[itemsep=0pt,topsep=-3pt]
    \item \textbf{Know} that an object will not speed up, slow down, or change direction on its own, and that it takes a net external force to change motion.
    \item \textbf{Know} that two interacting objects exert equal and opposite forces on each other.
    \item \textbf{Know} that balanced forces imply that net force (vector sum of all forces) is zero, and that unbalanced forces imply that net force is not zero.
\end{itemize}

\subsection*{Mathematical Formulas}

\begin{itemize}[topsep=-3pt,itemsep=0pt]
    \item \textbf{Know} that the base SI unit of force is the newton (N).
    \item \textbf{Know} that the base SI unit of mass is the kilogram (kg).
    \item \textbf{Know} that a newton (N) is equivalent to a kilogram times a meter per second squared ($\mathrm{kg\cdot m/s^2}$).
\end{itemize}

\subsection*{Mathematical Formulas}

\begin{itemize}[topsep=-3pt,itemsep=0pt]
    \item \textbf{Know} velocity as a function of time: $\displaystyle v_f = v_i + at$
    \item \textbf{Know} position as a function of time: $\displaystyle \Delta x = v_i t + \frac{1}{2}at^2$
    \item \textbf{Know} velocity as a function of position: $\displaystyle v_f^2 = v_i^2 + 2a\Delta x$
    \item \textbf{Know} average velocity to find displacement: $\displaystyle \Delta x = \frac{1}{2}(v_i + v_f)t$
    \item \textbf{Know} that hang time for an object dropped from rest from a given height is given by $\displaystyle t = \sqrt{\frac{2h}{g}}$.
    \item \textbf{Know} that the range of a horizontally launched projectile is given by $\Delta x = v_{ix} t$, where $t$ is the hang time.
\end{itemize}
